## 3. Р¤СѓРЅРєС†РёРѕРЅР°Р»СЊРЅС‹Рµ С‚СЂРµР±РѕРІР°РЅРёСЏ

### Р§Р°СЃС‚СЊ 1. РћР±РѕРіР°С‰РµРЅРёРµ РґР°РЅРЅС‹С… Р¤РЎРЎРџ
Р”Р»СЏ РєР°Р¶РґРѕР№ СЃС‚СЂРѕРєРё РІ С„Р°Р№Р»Рµ Р¤РЎРЎРџ РЅРµРѕР±С…РѕРґРёРјРѕ РЅР°Р№С‚Рё РґРµС‚Р°Р»РёР·Р°С†РёСЋ РІ СЃСѓРґРµР±РЅС‹С… СЂРµРµСЃС‚СЂР°С….

3. `re` вЂ” РґР»СЏ РѕС‡РёСЃС‚РєРё РЅРѕРјРµСЂРѕРІ РґРµР» (СЂРµРіСѓР»СЏСЂРЅС‹Рµ РІС‹СЂР°Р¶РµРЅРёСЏ).

### РњРѕРґСѓР»СЊ 1: РћС‡РёСЃС‚РєР° РЅРѕРјРµСЂРѕРІ РґРµР» (Regex)
РќРѕРјРµСЂР° РІ Р¤РЎРЎРџ (РЅР°РїСЂРёРјРµСЂ, `50MS0332#2-3674/2025#1`) Рё РІ СЂРµРµСЃС‚СЂР°С… СЃСѓРґРѕРІ (`2-3674/2025`) РІС‹РіР»СЏРґСЏС‚ РїРѕ-СЂР°Р·РЅРѕРјСѓ. РќР°Рј РЅСѓР¶РЅР° С„СѓРЅРєС†РёСЏ-РЅРѕСЂРјР°Р»РёР·Р°С‚РѕСЂ.

```python
import re

def normalize_case_number(case_str):
    if not case_str or str(case_str) == 'nan':
        return ""
    # РС‰РµРј РїР°С‚С‚РµСЂРЅ РІРёРґР° "С‡РёСЃР»Рѕ-С‡РёСЃР»Рѕ/РіРѕРґ" (РЅР°РїСЂРёРјРµСЂ 2-1234/2025)
    match = re.search(r'(\d+-\d+/\d{4})', str(case_str))
    if match:
        return match.group(1)
    return str(case_str).strip()
```

#### 3.2. РђР»РіРѕСЂРёС‚Рј "Р“Р»СѓР±РёРЅР° Р·Р°РґРѕР»Р¶РµРЅРЅРѕСЃС‚Рё" (DBF)

import os
from dbfread import DBF
import pandas as pd

def calculate_unprocessed_debt(account_id, cut_off_date, dbf_folder):
    # 1. РџРѕР»СѓС‡Р°РµРј СЃРїРёСЃРѕРє DBF С„Р°Р№Р»РѕРІ, СЃРѕСЂС‚РёСЂСѓРµРј РѕС‚ РЅРѕРІС‹С… Рє СЃС‚Р°СЂС‹Рј
    dbf_files = sorted([f for f in os.listdir(dbf_folder) if f.endswith('.dbf')], reverse=True)
    
    # 2. Р‘РµСЂРµРј СЃР°Р»СЊРґРѕ РёР· СЃР°РјРѕРіРѕ РїРѕСЃР»РµРґРЅРµРіРѕ С„Р°Р№Р»Р° (РЇРЅРІР°СЂСЊ 2026)
    last_dbf = DBF(os.path.join(dbf_folder, dbf_files[0]), encoding='cp866')
    
    target_saldo = 0
    for record in last_dbf:
        if str(record['T1']) == str(account_id):
            target_saldo = float(record['T39']) # РЎРђР›Р¬Р”Рћ РќРђ РљРћРќР•Р¦
            break
            
    if target_saldo <= 0: return 0, 0

    accumulated_charges = 0
    months_count = 0
    
    # 3. РРґРµРј РЅР°Р·Р°Рґ РїРѕ РјРµСЃСЏС†Р°Рј
    for file_name in dbf_files:
        # РР·РІР»РµРєР°РµРј РґР°С‚Сѓ РёР· РёРјРµРЅРё С„Р°Р№Р»Р° (РЅР°РїСЂ. 01_2026)
        file_date = parse_date_from_filename(file_name) 
        
        # Р•СЃР»Рё РґРѕС€Р»Рё РґРѕ РїСЂРѕСЃСѓР¶РµРЅРЅРѕРіРѕ РїРµСЂРёРѕРґР° - РЎРўРћРџ
        if cut_off_date and file_date <= cut_off_date:
            # Р•СЃР»Рё РЅР°С‡РёСЃР»РµРЅРёР№ РЅРµ С…РІР°С‚РёР»Рѕ, С‡С‚РѕР±С‹ РїРѕРєСЂС‹С‚СЊ СЃР°Р»СЊРґРѕ - Р±РµСЂРµРј РІСЃС‘ СЃР°Р»СЊРґРѕ (РїРѕ РўР—)
            return target_saldo, months_count
            
        # РС‰РµРј РЅР°С‡РёСЃР»РµРЅРёСЏ РІ СЌС‚РѕРј РјРµСЃСЏС†Рµ

        for record in month_dbf:
            if str(record['T1']) == str(account_id):
                charge = float(record['T33']) # РРўРћР“Рћ РќРђР§РРЎР›Р•РќРћ
                if charge > 0:
                    accumulated_charges += charge
                    months_count += 1
                break
        
        # Р•СЃР»Рё РЅР°С‡РёСЃР»РµРЅРёСЏ РїРѕРєСЂС‹Р»Рё СЃР°Р»СЊРґРѕ - РЎРўРћРџ
        if accumulated_charges >= target_saldo:
            return target_saldo, months_count

    return target_saldo, months_count
```

#### 2. Р›РѕРіРёРєР° Р§Р°СЃС‚Рё 1: SQL Join РґР»СЏ Р¤РЎРЎРџ

def enrich_fssp_via_sql(self):
    query = '''
        SELECT 
            f.*, 
            j.account_id, j.date_start, j.date_end, j.sum_main, j.sum_penalty, j.sum_duty, j.agent
        FROM raw_fssp f
        LEFT JOIN raw_judgments j ON f.case_number_normalized = j.case_number_normalized
    '''
    enriched_df = pd.read_sql(query, self.conn)
    return enriched_df
```
